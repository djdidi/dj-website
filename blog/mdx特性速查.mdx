```jsx live noInline
const Counter = (props) => {
  const [count, setCount] = React.useState<number>(0);
  return (
    <div>
      <h3 style={{
        background: 'darkslateblue',
        color: 'white',
        padding: 8,
        borderRadius: 4,
      }}>
        count = {count}
      </h3>
      <button
        onClick={() =>
          setCount(c => c + 1)
        }>
        Increment++
      </button>
    </div>
  );
};
render(<Counter />);
```
